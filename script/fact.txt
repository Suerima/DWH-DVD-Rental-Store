select i.film_id, f.rental_rate, f.rental_duration, sum(p.amount) SumReality, 
	  Sum(p.amount - (f.rental_rate / f.rental_duration) * DATEDIFF(day, r.rental_date, r.return_date)) Profit_early
from rental r
	inner join payment p on r.rental_id = p.rental_id
	inner join inventory i on i.inventory_id= r.inventory_id
	inner join film f on i.film_id = f.film_id
group by i.film_id, f.rental_rate, f.rental_duration
order by i.film_id, f.rental_rate, f.rental_duration

select r.rental_id, s.store_id,  i.film_id, f.rental_rate, f.rental_duration,
	  sum(p.amount) SumReality, 
	  Sum(p.amount - (f.rental_rate / f.rental_duration) * DATEDIFF(day, r.rental_date, r.return_date)) Profit_early
from rental r
	inner join payment p on r.rental_id = p.rental_id
	inner join inventory i on i.inventory_id= r.inventory_id
	inner join store s on s.store_id = i.store_id
	inner join film f on i.film_id = f.film_id
group by r.rental_id, s.store_id,  i.film_id, f.rental_rate, f.rental_duration
order by r.rental_id, s.store_id,  i.film_id, f.rental_rate, f.rental_duration
